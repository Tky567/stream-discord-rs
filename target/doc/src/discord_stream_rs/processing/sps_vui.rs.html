1/// H264 SPS VUI rewriter — ported from `SPSVUIRewriter.ts`.
2///
3/// Rewrites the VUI (Video Usability Information) section of an H264 SPS
4/// NALU to set `max_num_reorder_frames = 0`, which is required for real-time
5/// streaming over Discord.
6///
7/// Based on the WebRTC project reference implementation:
8/// <https://webrtc.googlesource.com/src/+/5f2c9278/common_video/h264/sps_vui_rewriter.cc>
9
10use super::annexb_rw::{AnnexBBitstreamReader, AnnexBBitstreamWriter};
11
12/// Rewrite the SPS NALU VUI timing info in-place and return the patched bytes.
13///
14/// `buffer` must start with the NAL unit header byte (NAL ref IDC + type 7).
15pub fn rewrite_sps_vui(buffer: &[u8]) -> Vec<u8> {
16    let mut reader = AnnexBBitstreamReader::new(&buffer[1..]);
17    let mut writer = AnnexBBitstreamWriter::new();
18
19    // Helpers (mirror TS aliases)
20    macro_rules! rb  { ($n:expr) => { reader.read_bits($n) } }
21    macro_rules! wb  { ($v:expr, $n:expr) => { writer.write_bits($v, $n) } }
22    macro_rules! ru  { ($n:expr) => { reader.read_unsigned($n) } }
23    macro_rules! wu  { ($v:expr, $n:expr) => { writer.write_unsigned($v, $n) } }
24    macro_rules! rue { () => { reader.read_ue() } }
25    macro_rules! wue { ($v:expr) => { writer.write_ue($v) } }
26    macro_rules! rse { () => { reader.read_se() } }
27    macro_rules! wse { ($v:expr) => { writer.write_se($v) } }
28
29    // NAL header byte (already consumed by the reader's slice offset 1)
30    wu!(buffer[0] as u32, 8);
31
32    let profile_idc = ru!(8);
33    wu!(profile_idc, 8);
34    let constraint_flags = ru!(8);
35    wu!(constraint_flags, 8);
36    let level_idc = ru!(8);
37    wu!(level_idc, 8);
38
39    let seq_parameter_set_id = rue!();
40    wue!(seq_parameter_set_id);
41
42    const HIGH_PROFILES: &[u32] = &[100, 110, 122, 244, 44, 83, 86, 118, 128, 138, 144];
43    if HIGH_PROFILES.contains(&profile_idc) {
44        let chroma_format_idc = rue!();
45        wue!(chroma_format_idc);
46
47        if chroma_format_idc == 3 {
48            let separate_colour_plane_flag = rb!(1);
49            wb!(separate_colour_plane_flag, 1);
50        }
51
52        let bit_depth_luma_minus8 = rue!();
53        wue!(bit_depth_luma_minus8);
54        let bit_depth_chroma_minus8 = rue!();
55        wue!(bit_depth_chroma_minus8);
56
57        let qpprime_y_zero_transform_bypass_flag = rb!(1);
58        wb!(qpprime_y_zero_transform_bypass_flag, 1);
59
60        let seq_scaling_matrix_present_flag = rb!(1);
61        wb!(seq_scaling_matrix_present_flag, 1);
62        if seq_scaling_matrix_present_flag != 0 {
63            let scaling_count = if chroma_format_idc != 3 { 8 } else { 12 };
64            for i in 0..scaling_count {
65                let seq_scaling_list_present_flag = rb!(1);
66                wb!(seq_scaling_list_present_flag, 1);
67                if seq_scaling_list_present_flag != 0 {
68                    let size = if i < 6 { 16 } else { 64 };
69                    let mut last_scale: i32 = 8;
70                    let mut next_scale: i32 = 8;
71                    for _ in 0..size {
72                        let delta = rse!();
73                        wse!(delta);
74                        next_scale = (last_scale + delta + 256) % 256;
75                        if next_scale != 0 {
76                            last_scale = next_scale;
77                        }
78                    }
79                    let _ = next_scale;
80                }
81            }
82        }
83    }
84
85    let log2_max_frame_num_minus4 = rue!();
86    wue!(log2_max_frame_num_minus4);
87
88    let pic_order_cnt_type = rue!();
89    wue!(pic_order_cnt_type);
90    if pic_order_cnt_type == 0 {
91        let log2_max_pic_order_cnt_lsb_minus4 = rue!();
92        wue!(log2_max_pic_order_cnt_lsb_minus4);
93    } else if pic_order_cnt_type == 1 {
94        let delta_pic_order_always_zero_flag = rb!(1);
95        wb!(delta_pic_order_always_zero_flag, 1);
96        let offset_for_non_ref_pic = rse!();
97        wse!(offset_for_non_ref_pic);
98        let offset_for_top_to_bottom_field = rse!();
99        wse!(offset_for_top_to_bottom_field);
100        let num_ref_frames_in_pic_order_cnt_cycle = rue!();
101        wue!(num_ref_frames_in_pic_order_cnt_cycle);
102        for _ in 0..num_ref_frames_in_pic_order_cnt_cycle {
103            let offset_for_ref_frame = rse!();
104            wse!(offset_for_ref_frame);
105        }
106    }
107
108    let max_num_ref_frames = rue!();
109    wue!(max_num_ref_frames);
110
111    let gaps_in_frame_num_value_allowed_flag = rb!(1);
112    wb!(gaps_in_frame_num_value_allowed_flag, 1);
113
114    let pic_width_in_mbs_minus1 = rue!();
115    wue!(pic_width_in_mbs_minus1);
116    let pic_height_in_map_units_minus1 = rue!();
117    wue!(pic_height_in_map_units_minus1);
118
119    let frame_mbs_only_flag = rb!(1);
120    wb!(frame_mbs_only_flag, 1);
121    if frame_mbs_only_flag == 0 {
122        let mb_adaptive_frame_field_flag = rb!(1);
123        wb!(mb_adaptive_frame_field_flag, 1);
124    }
125
126    let direct_8x8_inference_flag = rb!(1);
127    wb!(direct_8x8_inference_flag, 1);
128
129    let frame_cropping_flag = rb!(1);
130    wb!(frame_cropping_flag, 1);
131    if frame_cropping_flag != 0 {
132        let frame_crop_left_offset = rue!();
133        wue!(frame_crop_left_offset);
134        let frame_crop_right_offset = rue!();
135        wue!(frame_crop_right_offset);
136        let frame_crop_top_offset = rue!();
137        wue!(frame_crop_top_offset);
138        let frame_crop_bottom_offset = rue!();
139        wue!(frame_crop_bottom_offset);
140    }
141
142    // ---------------------------------------------------------------------------
143    // VUI parameters
144    // ---------------------------------------------------------------------------
145
146    let vui_parameters_present_flag = rb!(1);
147    wb!(1, 1); // always write VUI = present
148
149    if vui_parameters_present_flag == 0 {
150        // No VUI — write a minimal one
151        wb!(0, 2); // aspect_ratio_info_present_flag=0, overscan_info_present_flag=0
152        wb!(0, 1); // video_signal_type_present_flag=0
153        wb!(0, 5); // chroma_loc_info, timing, nal_hrd, vcl_hrd, pic_struct all 0
154        wb!(1, 1); // bitstream_restriction_flag=1
155        write_bitstream_restriction(&mut writer, max_num_ref_frames);
156    } else {
157        // Copy existing VUI, patching the bitstream restriction block
158        let aspect_ratio_info_present_flag = rb!(1);
159        wb!(aspect_ratio_info_present_flag, 1);
160        if aspect_ratio_info_present_flag != 0 {
161            let aspect_ratio_idc = ru!(8);
162            wu!(aspect_ratio_idc, 8);
163            if aspect_ratio_idc == 255 {
164                let sar_width = ru!(16);
165                wu!(sar_width, 16);
166                let sar_height = ru!(16);
167                wu!(sar_height, 16);
168            }
169        }
170
171        let overscan_info_present_flag = rb!(1);
172        wb!(overscan_info_present_flag, 1);
173        if overscan_info_present_flag != 0 {
174            let overscan_appropriate_flag = rb!(1);
175            wb!(overscan_appropriate_flag, 1);
176        }
177
178        // video_signal_type: read but strip (write 0)
179        let video_signal_type_present_flag = rb!(1);
180        wb!(0, 1); // strip video signal type
181        if video_signal_type_present_flag != 0 {
182            let _video_format = rb!(3);
183            let _video_full_range_flag = rb!(1);
184            let colour_description_present_flag = rb!(1);
185            if colour_description_present_flag != 0 {
186                let _colour_primaries = ru!(8);
187                let _transfer_characteristics = ru!(8);
188                let _matrix_coeffs = ru!(8);
189            }
190        }
191
192        let chroma_loc_info_present_flag = rb!(1);
193        wb!(chroma_loc_info_present_flag, 1);
194        if chroma_loc_info_present_flag != 0 {
195            let chroma_sample_loc_type_top_field = rue!();
196            wue!(chroma_sample_loc_type_top_field);
197            let chroma_sample_loc_type_bottom_field = rue!();
198            wue!(chroma_sample_loc_type_bottom_field);
199        }
200
201        let timing_info_present_flag = rb!(1);
202        wb!(timing_info_present_flag, 1);
203        if timing_info_present_flag != 0 {
204            let num_units_in_tick = ru!(32);
205            wu!(num_units_in_tick, 32);
206            let time_scale = ru!(32);
207            wu!(time_scale, 32);
208            let fixed_frame_rate_flag = rb!(1);
209            wb!(fixed_frame_rate_flag, 1);
210        }
211
212        let nal_hrd_parameters_present_flag = rb!(1);
213        wb!(nal_hrd_parameters_present_flag, 1);
214        if nal_hrd_parameters_present_flag != 0 {
215            copy_hrd_parameters(&mut reader, &mut writer);
216        }
217
218        let vcl_hrd_parameters_present_flag = rb!(1);
219        wb!(vcl_hrd_parameters_present_flag, 1);
220        if vcl_hrd_parameters_present_flag != 0 {
221            copy_hrd_parameters(&mut reader, &mut writer);
222        }
223
224        if nal_hrd_parameters_present_flag != 0 || vcl_hrd_parameters_present_flag != 0 {
225            let low_delay_hrd_flag = rb!(1);
226            wb!(low_delay_hrd_flag, 1);
227        }
228
229        let pic_struct_present_flag = rb!(1);
230        wb!(pic_struct_present_flag, 1);
231
232        let bitstream_restriction_flag = rb!(1);
233        wb!(1, 1); // always write bitstream_restriction = present
234
235        if bitstream_restriction_flag == 0 {
236            write_bitstream_restriction(&mut writer, max_num_ref_frames);
237        } else {
238            // Copy most fields but force reorder=0 and buffering=max_num_ref_frames
239            let motion_vectors_over_pic_boundaries_flag = rb!(1);
240            wb!(motion_vectors_over_pic_boundaries_flag, 1);
241            let max_bytes_per_pic_denom = rue!();
242            wue!(max_bytes_per_pic_denom);
243            let max_bits_per_mb_denom = rue!();
244            wue!(max_bits_per_mb_denom);
245            let log2_max_mv_length_horizontal = rue!();
246            wue!(log2_max_mv_length_horizontal);
247            let log2_max_mv_length_vertical = rue!();
248            wue!(log2_max_mv_length_vertical);
249            let _num_reorder_frames = rue!();
250            wue!(0); // force max_num_reorder_frames = 0
251            let _max_dec_frame_buffering = rue!();
252            wue!(max_num_ref_frames); // force max_dec_frame_buffering
253        }
254    }
255
256    wb!(1, 1); // rbsp_stop_one_bit
257    writer.flush_final();
258    writer.to_vec()
259}

261/// Write a default `bitstream_restriction()` block.
262fn write_bitstream_restriction(writer: &mut AnnexBBitstreamWriter, max_num_ref_frames: u32) {
263    writer.write_bits(1, 1); // motion_vectors_over_pic_boundaries_flag (default 1)
264    writer.write_ue(2); // max_bytes_per_pic_denom (default 2)
265    writer.write_ue(1); // max_bits_per_mb_denom (default 1)
266    writer.write_ue(16); // log2_max_mv_length_horizontal (default 16)
267    writer.write_ue(16); // log2_max_mv_length_vertical (default 16)
268    writer.write_ue(0); // max_num_reorder_frames = 0 (IMPORTANT!)
269    writer.write_ue(max_num_ref_frames); // max_dec_frame_buffering
270}

272/// Copy an `hrd_parameters()` block verbatim.
273fn copy_hrd_parameters(
274    reader: &mut AnnexBBitstreamReader<'_>,
275    writer: &mut AnnexBBitstreamWriter,
276) {
277    let cpb_cnt_minus1 = reader.read_ue();
278    writer.write_ue(cpb_cnt_minus1);
279    let bit_rate_scale = reader.read_bits(4);
280    writer.write_bits(bit_rate_scale, 4);
281    let cpb_size_scale = reader.read_bits(4);
282    writer.write_bits(cpb_size_scale, 4);
283    for _ in 0..=cpb_cnt_minus1 {
284        let bit_rate_value_minus1 = reader.read_ue();
285        writer.write_ue(bit_rate_value_minus1);
286        let cpb_size_value_minus1 = reader.read_ue();
287        writer.write_ue(cpb_size_value_minus1);
288        let cbr_flag = reader.read_bits(1);
289        writer.write_bits(cbr_flag, 1);
290    }
291    for _ in 0..4 {
292        // initial_cpb_removal_delay_length_minus1, cpb_removal_delay_length_minus1,
293        // dpb_output_delay_length_minus1, time_offset_length — each 5 bits
294        let v = reader.read_bits(5);
295        writer.write_bits(v, 5);
296    }
297}
