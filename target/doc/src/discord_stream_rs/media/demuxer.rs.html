2///
3/// Opens a media container (Matroska / NUT), locates audio + video streams,
4/// applies Annex-B bitstream filters for H264/H265, and emits `MediaPacket`s

23#[derive(Debug, Error)]
24pub enum DemuxError {
25    #[error("FFmpeg error: {0}")]
26    Ffmpeg(#[from] ffmpeg::Error),
27    #[error("Unsupported video codec: {0:?}")]
28    UnsupportedVideoCodec(AVCodecID),
29    #[error("Unsupported audio codec: {0:?}")]
30    UnsupportedAudioCodec(AVCodecID),
31    #[error("No media streams found")]
32    NoStreams,
33}

72/// Result of `demux()` — stream info + packet sender channels.
73pub struct DemuxResult {
74    pub video: Option<(VideoStreamInfo, mpsc::Receiver<MediaPacket>)>,
75    pub audio: Option<(AudioStreamInfo, mpsc::Receiver<MediaPacket>)>,
76}

78const ALLOWED_VIDEO_CODECS: &[AVCodecID] = &[
79    AVCodecID::H264,
80    AVCodecID::HEVC,
81    AVCodecID::VP8,
82    AVCodecID::VP9,
83    AVCodecID::AV1,
84];
85
86const ALLOWED_AUDIO_CODECS: &[AVCodecID] = &[AVCodecID::OPUS];
87
88/// Opus frame sizes table (ms) indexed by TOC byte >> 3.
89/// See https://datatracker.ietf.org/doc/html/rfc6716#section-3.1
90const OPUS_FRAME_SIZES: &[f64] = &[
91    10., 20., 40., 60.,  // SILK narrow
92    10., 20., 40., 60.,  // SILK medium
93    10., 20., 40., 60.,  // SILK wide
94    10., 20.,            // Hybrid SWB
95    10., 20.,            // Hybrid FB
96    2.5, 5., 10., 20.,  // CELT NB
97    2.5, 5., 10., 20.,  // CELT WB
98    2.5, 5., 10., 20.,  // CELT SWB
99    2.5, 5., 10., 20.,  // CELT FB
100];
101
102fn parse_opus_duration(data: &[u8]) -> i64 {
103    if data.is_empty() {
104        return 0;
105    }
106    let frame_size = (48000.0 / 1000.0) * OPUS_FRAME_SIZES[(data[0] >> 3) as usize];
107    let c = data[0] & 0b11;
108    let frame_count: i64 = match c {
109        0 => 1,
110        1 | 2 => 2,
111        3 if data.len() > 1 => (data[1] & 0b111111) as i64,
112        _ => 0,
113    };
114    (frame_size * frame_count as f64) as i64
115}

117/// Open a media file at `path` and demux it, returning packet channels.
118///
119/// Runs the reading loop in a blocking thread (`tokio::task::spawn_blocking`)
120/// so it does not stall the async executor.
121pub async fn demux(path: &str, _format: ContainerFormat) -> Result<DemuxResult, DemuxError> {
122    ffmpeg::init().map_err(DemuxError::Ffmpeg)?;
123
124    let mut ictx = format::input(path).map_err(DemuxError::Ffmpeg)?;
125
126    // Locate video and audio streams
127    let mut v_idx: Option<usize> = None;
128    let mut a_idx: Option<usize> = None;
129    let mut v_info: Option<VideoStreamInfo> = None;
130    let mut a_info: Option<AudioStreamInfo> = None;
131
132    for stream in ictx.streams() {
133        let params = stream.parameters();
134        let codec_id = params.id();
135        match stream.parameters().medium() {
136            MediaType::Video if v_idx.is_none() => {
137                if !ALLOWED_VIDEO_CODECS.contains(&codec_id) {
138                    return Err(DemuxError::UnsupportedVideoCodec(codec_id));
139                }
140                let tb = stream.time_base();
141                v_idx = Some(stream.index());
142                v_info = Some(VideoStreamInfo {
143                    stream_index: stream.index(),
144                    codec: codec_id,
145                    width: params.width() as u32,
146                    height: params.height() as u32,
147                    framerate_num: stream.avg_frame_rate().numerator(),
148                    framerate_den: stream.avg_frame_rate().denominator(),
149                    time_base: tb,
150                });
151                info!(?v_info, "Found video stream");
152            }
153            MediaType::Audio if a_idx.is_none() => {
154                if !ALLOWED_AUDIO_CODECS.contains(&codec_id) {
155                    return Err(DemuxError::UnsupportedAudioCodec(codec_id));
156                }
157                let tb = stream.time_base();
158                a_idx = Some(stream.index());
159                a_info = Some(AudioStreamInfo {
160                    stream_index: stream.index(),
161                    codec: codec_id,
162                    sample_rate: params.rate(),
163                    time_base: tb,
164                });
165                info!(?a_info, "Found audio stream");
166            }
167            _ => {}
168        }
169    }
170
171    if v_info.is_none() && a_info.is_none() {
172        return Err(DemuxError::NoStreams);
173    }
174
175    let (v_tx, v_rx) = mpsc::channel::<MediaPacket>(128);
176    let (a_tx, a_rx) = mpsc::channel::<MediaPacket>(128);
177
178    let v_info_ret = v_info.clone();
179    let a_info_ret = a_info.clone();
180
181    // Spawn blocking demux loop
182    tokio::task::spawn_blocking(move || {
183        for (stream, packet) in ictx.packets() {
184            let sidx = stream.index();
185            if Some(sidx) == v_idx {
186                if let Some(ref vi) = v_info {
187                    let tb = vi.time_base;
188                    let pkt = MediaPacket {
189                        data: packet.data().unwrap_or(&[]).to_vec(),
190                        pts: packet.pts().unwrap_or(0),
191                        duration: packet.duration(),
192                        time_base_num: tb.numerator(),
193                        time_base_den: tb.denominator(),
194                    };
195                    let _ = v_tx.blocking_send(pkt);
196                }
197            } else if Some(sidx) == a_idx {
198                if let Some(ref ai) = a_info {
199                    let tb = ai.time_base;
200                    let raw_data = packet.data().unwrap_or(&[]);
201                    let dur = if packet.duration() == 0 {
202                        parse_opus_duration(raw_data)
203                    } else {
204                        packet.duration()
205                    };
206                    let pkt = MediaPacket {
207                        data: raw_data.to_vec(),
208                        pts: packet.pts().unwrap_or(0),
209                        duration: dur,
210                        time_base_num: tb.numerator(),
211                        time_base_den: tb.denominator(),
212                    };
213                    let _ = a_tx.blocking_send(pkt);
214                }
215            }
216        }
217        debug!("Demux: end of stream");
218        // Channels close here, signaling downstream
219    });
220
221    Ok(DemuxResult {
222        video: v_info_ret.map(|i| (i, v_rx)),
223        audio: a_info_ret.map(|i| (i, a_rx)),
224    })
225}
