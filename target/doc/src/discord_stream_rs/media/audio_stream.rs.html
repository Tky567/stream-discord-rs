1/// Audio media stream — ported from `AudioStream.ts`.
2///
3/// Drains `MediaPacket` frames from a channel and forwards them to
4/// `WebRtcWrapper::send_audio_frame` with proper PTS-based timing.
5
6use crate::media::base_stream::{BaseMediaStream, MediaPacket, StreamSyncState};
7use crate::voice::webrtc::WebRtcWrapper;
8use std::sync::{Arc, Mutex};
9use tokio::sync::mpsc;
10use tokio::time::Instant;
11use tracing::debug;
12
13pub struct AudioStream {
14    base: BaseMediaStream,
15    webrtc: Arc<tokio::sync::Mutex<WebRtcWrapper>>,
16}
17
18impl AudioStream {
19    pub fn new(
20        webrtc: Arc<tokio::sync::Mutex<WebRtcWrapper>>,
21        no_sleep: bool,
22    ) -> (Self, Arc<Mutex<StreamSyncState>>) {
23        let (base, state) = BaseMediaStream::new(no_sleep);
24        (Self { base, webrtc }, state)
25    }
26
27    pub fn base_mut(&mut self) -> &mut BaseMediaStream {
28        &mut self.base
29    }
30
31    /// Drive the stream from an async receiver.  Returns when the channel closes
32    /// or `stop_rx` fires.
33    pub async fn run(
34        &mut self,
35        mut rx: mpsc::Receiver<MediaPacket>,
36        mut stop_rx: tokio::sync::oneshot::Receiver<()>,
37    ) {
38        let mut start_time: Option<Instant> = None;
39        let mut start_pts: Option<f64> = None;
40
41        loop {
42            tokio::select! {
43                biased;
44                _ = &mut stop_rx => {
45                    debug!("AudioStream stopped");
46                    break;
47                }
48                pkt = rx.recv() => {
49                    match pkt {
50                        None => break,
51                        Some(packet) => {
52                            let webrtc = self.webrtc.clone();
53                            self.base
54                                .process_packet(
55                                    &packet,
56                                    &mut start_time,
57                                    &mut start_pts,
58                                    |data, ft| async move {
59                                        let mut w = webrtc.lock().await;
60                                        let _ = w.send_audio_frame(&data, ft).await;
61                                    },
62                                )
63                                .await;
64                        }
65                    }
66                }
67            }
68        }
69
70        self.base.mark_ended();
71    }
72}
