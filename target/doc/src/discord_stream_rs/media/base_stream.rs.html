1/// Base media stream — ported from `BaseMediaStream.ts`.
2///
3/// Implements the PTS-based timing / sync logic for draining encoded frames
4/// into the Discord WebRTC pipeline at the correct wall-clock rate.
5
6use std::sync::{Arc, Mutex};
7use tokio::time::{sleep, Duration, Instant};
8use tracing::{debug, warn};
9
10// ---------------------------------------------------------------------------
11// Packet (mirrors `node-av` Packet)
12// ---------------------------------------------------------------------------
13
14/// A decoded/encoded media frame packet, analogous to the `Packet` type from
15/// `node-av`.
16#[derive(Debug, Clone)]
17pub struct MediaPacket {
18    pub data: Vec<u8>,
19    /// Presentation timestamp (raw codec time units).
20    pub pts: i64,
21    /// Frame duration (raw codec time units).
22    pub duration: i64,
23    /// Timebase: `(num, den)`.  Frametime = `duration * num / den * 1000` ms.
24    pub time_base_num: i32,
25    pub time_base_den: i32,
26}
27
28impl MediaPacket {
29    /// Compute the frame duration in milliseconds.
30    pub fn frametime_ms(&self) -> f64 {
31        (self.duration as f64 / self.time_base_den as f64) * self.time_base_num as f64 * 1000.0
32    }
33
34    /// Compute PTS in milliseconds.
35    pub fn pts_ms(&self) -> f64 {
36        (self.pts as f64 / self.time_base_den as f64) * self.time_base_num as f64 * 1000.0
37    }
38}

44/// Mutable PTS state shared via `Arc<Mutex<>>` across audio and video streams
45/// so one can check the other's progress.
46#[derive(Debug, Default)]
47pub struct StreamSyncState {
48    pub pts_ms: Option<f64>,
49    pub ended: bool,
50}

56/// Tolerance in ms within which two streams are considered in sync.
57const SYNC_TOLERANCE_MS: f64 = 20.0;
58
59pub struct BaseMediaStream {
60    pub sync: bool,
61    pub no_sleep: bool,
62    sync_tolerance: f64,
63
64    /// Our own PTS/ended state (shared with peer stream).
65    state: Arc<Mutex<StreamSyncState>>,
66    /// The peer stream's state (for A/V sync decisions).
67    peer_state: Option<Arc<Mutex<StreamSyncState>>>,
68}
69
70impl BaseMediaStream {
71    pub fn new(no_sleep: bool) -> (Self, Arc<Mutex<StreamSyncState>>) {
72        let state = Arc::new(Mutex::new(StreamSyncState::default()));
73        let stream = Self {
74            sync: true,
75            no_sleep,
76            sync_tolerance: SYNC_TOLERANCE_MS,
77            state: state.clone(),
78            peer_state: None,
79        };
80        (stream, state)
81    }
82
83    /// Link this stream to a peer (e.g. video ↔ audio) for AV sync.
84    pub fn set_sync_stream(&mut self, peer: Arc<Mutex<StreamSyncState>>) {
85        // Guard against mutual sync
86        assert!(
87            !Arc::ptr_eq(&self.state, &peer),
88            "Cannot sync a stream with itself"
89        );
90        self.peer_state = Some(peer);
91    }
92
93    pub fn sync_tolerance(&self) -> f64 {
94        self.sync_tolerance
95    }
96
97    pub fn set_sync_tolerance(&mut self, ms: f64) {
98        if ms >= 0.0 {
99            self.sync_tolerance = ms;
100        }
101    }
102
103    fn pts_delta(&self) -> Option<f64> {
104        let our_pts = self.state.lock().ok()?.pts_ms?;
105        let peer_pts = self.peer_state.as_ref()?.lock().ok()?.pts_ms?;
106        Some(our_pts - peer_pts)
107    }
108
109    fn peer_active(&self) -> bool {
110        self.peer_state
111            .as_ref()
112            .and_then(|p| p.lock().ok())
113            .map(|s| !s.ended)
114            .unwrap_or(false)
115    }
116
117    fn is_ahead(&self) -> bool {
118        self.peer_active()
119            && self
120                .pts_delta()
121                .map(|d| d > self.sync_tolerance)
122                .unwrap_or(false)
123    }
124
125    fn is_behind(&self) -> bool {
126        self.peer_active()
127            && self
128                .pts_delta()
129                .map(|d| d < -self.sync_tolerance)
130                .unwrap_or(false)
131    }
132
133    /// Update our PTS (in ms) after sending a frame.
134    pub fn update_pts(&self, pts_ms: f64) {
135        if let Ok(mut s) = self.state.lock() {
136            s.pts_ms = Some(pts_ms);
137        }
138    }
139
140    pub fn mark_ended(&self) {
141        if let Ok(mut s) = self.state.lock() {
142            s.ended = true;
143        }
144    }
145
146    /// Process one packet: call `send_frame`, then sleep as needed to maintain
147    /// wall-clock timing.  Mirrors `_write()` in `BaseMediaStream.ts`.
148    pub async fn process_packet<F, Fut>(
149        &mut self,
150        packet: &MediaPacket,
151        start_time: &mut Option<Instant>,
152        start_pts: &mut Option<f64>,
153        send_frame: F,
154    ) where
155        F: FnOnce(Vec<u8>, f64) -> Fut,
156        Fut: std::future::Future<Output = ()>,
157    {
158        let frametime = packet.frametime_ms();
159        let pts_ms = packet.pts_ms();
160
161        let t0 = Instant::now();
162        send_frame(packet.data.clone(), frametime).await;
163        let send_dur = t0.elapsed().as_secs_f64() * 1000.0;
164
165        self.update_pts(pts_ms);
166
167        let ratio = send_dur / frametime;
168        debug!(
169            pts = pts_ms,
170            frame_size = packet.data.len(),
171            send_ms = send_dur,
172            frametime,
173            "Frame sent ({:.1}% of frametime)",
174            ratio * 100.0
175        );
176        if ratio > 1.0 {
177            warn!(
178                frame_size = packet.data.len(),
179                send_ms = send_dur,
180                frametime,
181                "Frame takes too long to send ({:.1}% of frametime)",
182                ratio * 100.0
183            );
184        }
185
186        *start_time = start_time.or(Some(t0));
187        *start_pts = start_pts.or(Some(pts_ms));
188
189        let elapsed_ms = start_time.map(|s| s.elapsed().as_secs_f64() * 1000.0).unwrap_or(0.0);
190        let sleep_ms = (pts_ms - start_pts.unwrap_or(pts_ms) + frametime - elapsed_ms).max(0.0);
191
192        if self.no_sleep || sleep_ms == 0.0 {
193            // no sleep
194        } else if self.sync && self.is_behind() {
195            debug!("Stream is behind; skipping sleep for this frame");
196            *start_time = None;
197            *start_pts = None;
198        } else if self.sync && self.is_ahead() {
199            loop {
200                debug!("Stream is ahead; waiting {}ms", frametime);
201                sleep(Duration::from_secs_f64(frametime / 1000.0)).await;
202                if !self.is_ahead() {
203                    break;
204                }
205            }
206            *start_time = None;
207            *start_pts = None;
208        } else {
209            sleep(Duration::from_secs_f64(sleep_ms / 1000.0)).await;
210        }
211    }
